    <script>
        // Obtenemos el id del usuario desde la URL
        const params = new URLSearchParams(window.location.search);
        const idUsuario = params.get("id_usuario");

        if (!idUsuario) {
            window.location.href = "home.html";
        }

        document.addEventListener("DOMContentLoaded", async () => {
            try {
                // Llamamos a un servlet para obtener los datos del usuario
                const response = await fetch("usuario?id_usuario=" + idUsuario, {
                    method: "GET",
                    credentials: "include"
                });

                if (!response.ok) {
                    window.location.href = "home.html";
                    return;
                }

                const data = await response.json();

                if (!data || !data.nombre) {
                    window.location.href = "home.html";
                    return;
                }

                document.getElementById("nombre_otro").textContent = data.nombre;
                document.getElementById("correo_otro").textContent = data.correo;

            } catch (error) {
                window.location.href = "home.html";
            }
        });
    </script>
